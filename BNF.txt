<cline> ::= <command_create> | <command_add> | <command_set> | <command_list> | <command_calculate> | <command_dump>

<command_create> ::= "create" <phone> | "create" <part>
<command_add> ::= "add" <subpart>
<command_set> ::= "set" <setcost>
<command_list> ::= "list" <listing>
<command_calculate> ::= "calculate" <calc>
<command_dump> ::= "dump" <ident>

<phone> ::= <ident> <antenna> <keypad> <camera> <proccessor>
<antenna> ::= "Wi-fi 6" | "Wi-fi 7"
<keypad> ::= "German" | "Swedish" | "American"
<camera> ::= "Basic" | "Pro" | "Pro Ultra"
<proccessor> ::= "X6" | "X7" | "X8"
<model> ::= "model" <qname> "to" <ident>
<part> ::= "part" <qname> "to" <mpath>
<subpart> ::= "subpart" <qname> "under" <ppath>

<setcost> ::= "cost" <ppath> "=" <number>
<listing> ::= "partsin" <mpath>
<calc> ::= "costof" <target>

<target> ::= <mpath> | <ppath>
<mpath> ::= <ident> "/" <qname>
<ppath> ::= <mpath> "/" <segs>
<segs> ::= <qname> | <qname> "/" <segs>

<qname> ::= "\"" <wordseq> "\""
<wordseq> ::= <word> | <word> " " <wordseq>
<word> ::= <alphanum> | <alphanum> <word>

<ident> ::= <letter> | <letter> <ident>
<alphanum> ::= <letter> | <digit>
<letter> ::= [a-z] | [A-Z]
<digit> ::= [0-9]

<number> ::= <digits> | <digits> "." <digits>
<digits> ::= <digit> | <digit> <digits>
